package main

import "fmt"

func main() {
	n := 0
	c := 0
	fmt.Println("输入n：")
	fmt.Scan(&n)
	w := make([]int, n)
	fmt.Println("输入w")
	for i := 0; i < n; i++ {
		fmt.Scan(&w[i])
	}
	fmt.Println("输入v")
	v := make([]int, n)
	for i := 0; i < n; i++ {
		fmt.Scan(&v[i])
	}
	fmt.Println("输入c：")
	fmt.Scan(&c)
	bag(w, v, c)

}

func bag(w, v []int, c int) {
	dp := make([]int, c+1)
	for i := 0; i < len(w); i++ {
		for j := c; j >= w[i]; j-- {
			dp[j] = max(dp[j], dp[j-w[i]]+v[i])
			3
		}
	}
	fmt.Println(dp[c])
}

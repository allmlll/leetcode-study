package main

import (
	"fmt"
)

func main() {
	n := 0
	c := 0
	fmt.Println("输入n：")
	fmt.Scan(&n)
	w := make([]int, n)
	fmt.Println("输入w")
	for i := 0; i < n; i++ {
		fmt.Scan(&w[i])
	}
	fmt.Println("输入v")
	v := make([]int, n)
	for i := 0; i < n; i++ {
		fmt.Scan(&v[i])
	}
	fmt.Println("输入c：")
	fmt.Scan(&c)
	bag(w, v, c)
}
func bag(w, v []int, c int) {
	res := 0
	for i := len(v) - 1; i > 0; i-- {
		if w[i] <= c {
			c -= w[i]
			res += v[i]
		}
	}
	fmt.Println(res)
}
